- Para vitalicio: duração vitalícia (sem expiração).

  POST https://api.mercadopago.com/checkout/preferences

  - items: [{title, quantity:1, currency_id:"BRL", unit_price: preço fixo}]

  - back_urls:
      success: {APP_URL}/pagamento/sucesso
      failure: {APP_URL}/pagamento/erro
      pending: {APP_URL}/pagamento/pendente
